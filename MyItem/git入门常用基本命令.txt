git clone "地址"  //从git上下载代码到本地
git status //查看文件是否被更改状态，会显示add，delete，modify等信息
git add 文件名或者文件夹名 //添加指定的文件或者文件夹进上传队列
git add . //把文件夹内所有未被添加的文件，全部加入上传队列
git rm 文件名或者文件夹名 //删除指定的文件或者文件夹进上传队列
git mv -f 旧名字 新名字 //重命名文件
git commit -m "(提交内容说明)"  //提交更改内容说明
git push //需要输入git的账号名和密码，将更改的内容上传到仓库
git pull //同步当前仓库的代码
git commit -a -m "(提交内容说明)"  //将当前项目中所有更改未提交的全部提交，并带上更改内容说明
